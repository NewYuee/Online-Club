public class ClubAppTable implements Serializable{

    private int code; //
    private String msg; //
    private ApplicationJson[] data; //
    private int count; //

    public int getCode(){
        return code;
    }

    public void setCode(int code){
        this.code = code;
    }

    public String getMsg(){
        return msg;
    }

    public void setMsg(String msg){
        this.msg = msg;
    }

    public ApplicationJson[] getData(){
        return data;
    }

    public void setData(ApplicationJson[] data){
        this.data = data;
    }

    public int getCount(){
        return count;
    }

    public void setCount(int count){
        this.count = count;
    }

}
